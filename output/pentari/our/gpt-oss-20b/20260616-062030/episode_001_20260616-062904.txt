def extract_state(game_history):
    # Determine if the postcard has been read
    if ("read postcard" in game_history.lower()) or ("postcard read" in game_history.lower()):
        postcard_read = True
    else:
        postcard_read = False
    # Check if sword is in inventory
    if "sword" in game_history.lower():
        sword_in_inventory = True
    else:
        sword_in_inventory = False
    # Check if dresser has been examined
    dresser_examined = "examined dresser" in game_history.lower()
    # Check if any drawer was opened
    drawer_opened = any(f"open drawer{n}" in game_history.lower() for n in range(1,5))
    # Determine if door was found/attempted
    door_attempted = any(cmd in game_history.lower() for cmd in ["open door", "examine door", "search door"])
    # Assemble summary
    parts = []
    if postcard_read:
        parts.append("Postcard read")
    if sword_in_inventory:
        parts.append("Sword in inventory")
    if dresser_examined:
        parts.append("Dresser examined")
    if drawer_opened:
        parts.append("Drawer opened")
    if door_attempted:
        parts.append("Door inspected")
    if not parts:
        return "No significant progress."
    return "; ".join(parts) + "."
